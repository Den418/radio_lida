# Telegram-клиент — pyrofork, а НЕ pyrogram!
#
#   Почему не pyrogram 2.0.106?
#   py-tgcalls импортирует GroupcallForbidden из pyrogram.errors —
#   в официальном pyrogram этого класса больше нет (убрали в 2.0.x).
#   pyrofork — поддерживаемый форк pyrogram, который его сохранил.
#   Устанавливается как модуль "pyrogram" → весь код остаётся без изменений.
pyrofork>=2.3.0

# TgCrypto (ускорение шифрования) — НАМЕРЕННО НЕ включён сюда!
#
#   Официальный пакет TgCrypto заброшен (последний релиз — 2018 год) и не
#   имеет готовых сборок под современный Python. pip пытается собрать его
#   из исходников через компилятор C++, которого на Windows обычно нет —
#   сборка падает, и ВЕСЬ pip install обрывается, не установив вообще ничего.
#
#   Без TgCrypto pyrofork просто использует более медленную встроенную
#   Python-реализацию шифрования (PyAES) — для музыкального бота разница
#   незаметна. Ничего не сломается, будет лишь предупреждение в логах.
#
#   Хочешь всё же поставить TgCrypto? Сначала поставь Visual C++ Build Tools:
#   https://visualstudio.microsoft.com/visual-cpp-build-tools/
#   Затем: pip install TgCrypto

# Голосовые чаты
# ВАЖНО: py-tgcalls (с дефисом), НЕ pytgcalls (без дефиса)
#   pytgcalls (старый) → tgcalls  → нет колёс для Python 3.10+ и Windows
#   py-tgcalls (новый) → ntgcalls → есть колёса для всех платформ
py-tgcalls>=2.0.0
